% Switch Statement

% From [this TeX Stack Exchange answer](https://tex.stackexchange.com/a/187710/64441).
\usepackage{ifthen}

\newcommand{\ifequals}[3]{
  \ifthenelse{\equal{#1}{#2}}{#3}{}
}

% Dummy stuff, so \renewcommand has something to overwrite...
\newcommand{\case}[2]{
  #1 #2
} 

\newenvironment{switch}[1]{
  \renewcommand{\case}{\ifequals{#1}}
}{}

%-----------------------------------------------------------
% Grid

\newlength{\size}
\newcommand{\gogrid}[1][1]{
  \setlength{\size}{\dimexpr #1cm / 18 \relax} % chktex 1

  \draw[step=\size] (0, 0) grid (#1, #1);
  
  \boardoutline{#1}
  
  \coordinates{#1} % chktex 1
}

\newcounter{coordinatecounter}
\setcounter{coordinatecounter}{1}
\newcommand{\coordinates}[1]{
  \foreach [count=\i] \l in {A, B, ..., T} {
    \ifthenelse{\equal{\l}{I}}{}{
      \draw[
        xshift=-0.5cm,
        yshift=(\thecoordinatecounter - 1) * 0.555cm,
      ] node {\l};
      \draw[
        xshift=(\thecoordinatecounter - 1) * 0.555cm,
        yshift=#1cm + 0.5cm,
      ] node {\i };

      \stepcounter{coordinatecounter}
    }
  }
  \setcounter{coordinatecounter}{0}
}

\newcommand{\boardoutline}[1]{
  % Top Horizontal
  \draw[black, line width = 1]
    (0, #1) -- (\dimexpr #1cm + 0.01cm, #1); % chktex 1 chktex 8
  % Right Vertical
  \draw[black, line width = 1] 
    (#1, 0) -- (#1, #1); % chktex 8
  % Bottom Horizontal
  \draw[black, line width = 1]
    (0, 0) -- (\dimexpr #1cm + 0.01cm, 0); % chktex 1 chktex 8
  % Left Vertical
  \draw[black, line width = 1]
    (0, 0) -- (0, #1); % chktex 8
}

%-----------------------------------------------------------
% Stones

\newcommand{\drawstone}[3]{
  \begin{switch}{#1}
    \case{black}{
      \draw[draw = white, line width = 0.1mm, fill = black]
        (#2, #3) circle [radius = 0.2575cm]; % chktex 1
    }
    \case{white}{
      \draw[draw = black, line width = 0.1mm, fill = white, smooth]
        (#2, #3) circle [smooth, radius = 0.25cm]; % chktex 1
    }
  \end{switch}
}

\newcounter{movecounter}
\setcounter{movecounter}{1}
\newcommand{\drawmove}[3]{
  \begin{switch}{#1}
    % TODO: Change fontsize if the number goes beyond 99.
    \case{black}{
      \draw[draw = white, line width = 0.1mm, fill = black]
        (#2, #3) circle [radius = 0.2575cm] % chktex 1
        node[color = white] {\themovecounter};
    }
    \case{white}{
      \draw[draw = black, line width = 0.1mm, fill = white, smooth]
        (#2, #3) circle [smooth, radius = 0.25cm] % chktex 1
        node {\themovecounter};
    }
    \stepcounter{movecounter}
  \end{switch}
}

\newenvironment{goban}{
  \begin{tikzpicture}
    \gogrid[10]
}{
    \setcounter{movecounter}{1}
  \end{tikzpicture}
}
